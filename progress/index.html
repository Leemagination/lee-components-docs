进度条 Progress

展示一些事件的进度情况

基础

演示
百分比: 10
10%
100%
30%
50%
70%
vue文件
<script setup>
import { LeeProgress, LeeSlider } from "lee-vue3-components";
import { ref } from "vue";

const percent1 = ref(10)
const percent2 = ref(30)
const percent3 = ref(50)
const percent4 = ref(70)
</script>

<template>
  <div>
    <div>百分比: {{ percent1 }}</div>
    <LeeSlider v-model:value="percent1"></LeeSlider>
    <LeeProgress :percent="percent1"></LeeProgress>
    <LeeProgress :percent="100" status="normal"></LeeProgress>
    <LeeProgress :percent="percent2" status="success"></LeeProgress>
    <LeeProgress :percent="percent3" status="warning"></LeeProgress>
    <LeeProgress :percent="percent4" status="error"></LeeProgress>
    <LeeProgress type="circle" :percent="percent1"></LeeProgress>
    <LeeProgress type="circle" :percent="percent2" status="success"></LeeProgress>
    <LeeProgress type="circle" :percent="percent3" status="warning"></LeeProgress>
    <LeeProgress type="circle" :percent="percent4" status="error"></LeeProgress>
  </div>
</template>

10%
10%
10%
vue文件
<script setup>
import { LeeProgress, LeeSlider } from "lee-vue3-components";
import { ref } from "vue";

const percent1 = ref(10)

</script>

<template>
  <div>
    <div>百分比: {{ percent1 }}</div>
    <LeeSlider v-model:value="percent1"></LeeSlider>
    <div>进度条宽度</div>
    <LeeProgress :percent="percent1" :width="16"></LeeProgress>
    <div>圆圈半径和宽度</div>
    <LeeProgress type="circle" :percent="percent1" :radius="30" :width="12"></LeeProgress>
    <div>不展示数值</div>
    <LeeProgress :percent="percent1" :width="16" :show-value="false"></LeeProgress>
    <LeeProgress type="circle" :percent="percent1" :show-value="false" :width="12"></LeeProgress>
    <div>渐变</div>
    <LeeProgress
      :percent="percent1"
      :width="16"
      :show-value="false"
      gradient="linear-gradient(to right,yellow,pink)">
    </LeeProgress>
    <div>定制颜色</div>
    <LeeProgress :percent="percent1" status="success" :color-config="{success:'pink'}"></LeeProgress>
    <LeeProgress :percent="percent1" status="warning" :color-config="{warning:'purple'}"></LeeProgress>
    <LeeProgress :percent="percent1" status="error" :color-config="{error:'gray'}"></LeeProgress>
  </div>
</template>

API

Progress props

名称类型默认值说明
typeline | circleline进度条展示类型
percentnumber0进度条百分比
status'normal' | 'success' | 'error' | 'warning'undefined进度条状态颜色
widthnumber8进度条宽度
radiusnumber50圆圈进度条半径
showValuebooleantrue是否展示进度百分比
gradientstringnull渐变颜色配置
colorConfig{[p:status]:string}{}状态颜色配置
