信息 Message

浮层展示的信息,让用户更容易接收到重要的信息。命令式用法

基础

演示
vue文件
<script setup>
import { useMessage, LeeButtonGroup, LeeButton } from "lee-vue3-components";

function openMessage(type) {
  useMessage({
    content: "这是一段信息,biubiubiu",
    type
  })
}
</script>

<template>
  <div>
    <LeeButtonGroup>
      <LeeButton @click="openMessage()">普通</LeeButton>
      <LeeButton @click="openMessage('info')">info信息</LeeButton>
      <LeeButton @click="openMessage('success')">success信息</LeeButton>
      <LeeButton @click="openMessage('warning')">warning信息</LeeButton>
      <LeeButton @click="openMessage('error')">error信息</LeeButton>
    </LeeButtonGroup>
  </div>
</template>

其他配置

演示
vue文件
<script setup>
import { setMessageConfig, useMessage, LeeButtonGroup, LeeButton, clearAllMessage } from "lee-vue3-components";

function openDurationMessage() {
  useMessage({
    duration: 5000,
    content: "这里展示5s",
  })
}

function openClosableMessage() {
  useMessage({
    closable: true,
    duration: 0,
    content: "一段信息,hello world,不会自动消失",
  })
}

function configMessage() {
  setMessageConfig({
    maxItem: 10,
    placement: 'bottom'
  })
  useMessage({
    closable: true,
    type: 'error',
    content: "一段信息,hello world",
  })
}

function closeMessage() {
  const { close } = useMessage({
    duration: 0,
    content: '设置一个duration为0的信息,手工5s后关闭'
  })
  setTimeout(() => {
    close()
  }, 5000)
}
</script>

<template>
  <div>
    <LeeButtonGroup>
      <LeeButton @click="openDurationMessage()">持续5s</LeeButton>
      <LeeButton @click="openClosableMessage()">可关闭的</LeeButton>
      <LeeButton @click="closeMessage()">人工关闭信息</LeeButton>
      <LeeButton @click="configMessage()">全局配置</LeeButton>
      <LeeButton @click="clearAllMessage()">清除所有信息</LeeButton>
    </LeeButtonGroup>
  </div>
</template>

API

useMessage params

名称类型默认值说明
contentstring''信息内容
durationnumber3000信息停留时间(ms)
typeinfo|success|warning|errorundefined信息类型
closablebooleanfalse是否展示关闭按钮

setMessageConfig params

名称类型默认值说明
maxItemnumber5最多展示信息条数
durationnumber3000信息停留时间(ms)
placementPlacementTypeundefined信息展示位置

PlacementType类型

值说明
top上方居中
top-left上方左侧
top-right上方右侧
bottom底部居中
bottom-left底部左侧
bottom-right底部右侧
