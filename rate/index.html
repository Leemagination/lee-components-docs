vue文件
<script setup>
import { LeeRate } from "lee-vue3-components";
import { ref } from "vue";

const rateValue = ref(3)
</script>

<template>
  <div>
    <div>分值: {{ rateValue }}</div>
    <LeeRate v-model:value="rateValue"></LeeRate>
    <div>个数</div>
    <LeeRate v-model:value="rateValue" :count="10"></LeeRate>
  </div>
</template>

vue文件
<script setup>
import { LeeRate } from "lee-vue3-components";
import { ref } from "vue";

const rateValue = ref(3)
const rateValue2 = ref(1.5)
</script>

<template>
  <div>
    <div>内置图标</div>
    <LeeRate v-model:value="rateValue" icon-type="fire"></LeeRate>
    <br>
    <LeeRate v-model:value="rateValue" icon-type="heart"></LeeRate>
    <br>
    <LeeRate v-model:value="rateValue" icon-type="cup"></LeeRate>
    <div>可重置的</div>
    <LeeRate v-model:value="rateValue" clearable></LeeRate>
    <div>可半分</div>
    <LeeRate v-model:value="rateValue2" allow-half></LeeRate>
    <div>自定义颜色</div>
    <LeeRate v-model:value="rateValue2" allow-half color="pink"></LeeRate>
    <div>可自定字符</div>
    <LeeRate v-model:value="rateValue2" allow-half>囍</LeeRate>
    <div>只可读</div>
    <LeeRate v-model:value="rateValue2" allow-half readonly></LeeRate>
  </div>
</template>

API

Rate props

名称类型默认值说明
valuenumber0评分的值
iconTypestar | fire | heart | cupstar评分内置图标
countnumber5评分的个数
clearablebooleanfalse评分是否可点击清空
allowHalfbooleanfalse评分是否能打半分
readonlybooleanfalse评分是否只读
colorstring#ffb838评分展示的颜色
